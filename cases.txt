Name: Element
Input:
    div
Output:
    <div></div>


Name: Nested elements
Input:
    html
        head
        body
            header
            footer
Output:
    <html>
        <head></head>
        <body>
            <header></header>
            <footer></footer>
        </body>
    </html>


Name: Shorthand syntax for nesting
Input:
    header
        div > ul
            li
            li
        p
    footer
Output:
    <header>
        <div>
            <ul>
                <li></li>
                <li></li>
            </ul>
        </div>
        <p></p>
    </header>
    <footer></footer>


Name: Element with text
Input:
    div | Some text
Output:
    <div>Some text</div>


Name: Text nodes
Input:
    p
        | Text
        | and another text
Output:
    <p>
        Text
        and another text
    </p>


Name: Mixed content
Input:
    p
        | Text
        em | emphasis
        | and another text
Output:
    <p>
        Text
        <em>emphasis</em>
        and another text
    </p>


Name: Multi-line (block) text
Input:
    |
        Text
            indented text
        and another text
Output:
    Text
        indented text
    and another text


Name: Attributes
Input:
    a href="http://website.example/" target="_blank"
Output:
    <a href="http://website.example/" target="_blank"></a>


Name: Elements without end tag (void elements)
Input:
    img src="1.jpg" alt="Some text"
    br
Output:
    <img src="1.jpg" alt="Some text"/>
    <br/>


Name: Boolean (empty) attributes
Input:
    option selected
Output:
    <option selected></option>


Name: ID attribute
Input:
    span#foo
Output:
    <span id="foo"></span>


Name: Class attribute
Input:
    span.foo
Output:
    <span class="foo"></span>


Name: Several classes
Input:
    span.foo.bar
Output:
    <span class="foo bar"></span>


Name: ID and class
Input:
    span#foo.bar.baz
Output:
    <span id="foo" class="bar baz"></span>


Name: Overriding ID attribute
Input:
    span#foo id="bar"
Output:
    <span id="bar"></span>


Name: Overriding class attribute
Input:
    span.foo class="bar"
Output:
    <span class="foo bar"></span>


Name: Default element is div
Input:
    #foo
    .bar
    #foo.bar.baz
Output:
    <div id="foo"></div>
    <div class="bar"></div>
    <div id="foo" class="bar baz"></div>


Name: Long attributes
Input:
    input type="text"
        & name="title"
        & maxlength="10"
        & value="Some text"
Output:
    <input type="text" name="title" maxlength="10" value="Some text"/>


Name: Comment
Input:
    ! Text of comment
Output:
    <!-- Text of comment -->


Name: Hidden comment
Input:
    div
    !! Some internal comment
    p
Output:
    <div></div>
    <p></p>


Name: Special DOCTYPE comment
Input:
    !DOCTYPE HTML
Output:
    <!DOCTYPE HTML>


Name: Special XML comment
Input:
    !xml version="1.0" encoding="UTF-8"
Output:
    <?xml version="1.0" encoding="UTF-8"?>


Name: Multi-line (block) comment
Input:
    !
        Comment
        and another line
Output:
    <!--
        Comment
        and another line
    -->


Name: Multi-line (block) hidden comment
Input:
    div
    !!
        Comment
        and another line
    p
Output:
    <div></div>
    <p></p>


Name: Mixed inline and multi-line text
Input:
    | Text
        First line
        Second line
Output:
    Text
        First line
        Second line


Name: Mixed inline and multi-line comment
Input:
    ! Comment
        First line
        Second line
Output:
    <!-- Comment
        First line
        Second line
    -->


Name: Mixed inline and multi-line hidden comment
Input:
    div
    !! Comment
        First line
        Second line
    p
Output:
    <div></div>
    <p></p>
